**app.py:**
```python
from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy
from flask_marshmallow import Marshmallow
from config import Config
from models import Book
from schemas import BookSchema
from repository import BookRepository
from services import BookService
from utils import Logger

db = SQLAlchemy()
ma = Marshmallow()
logger = Logger()

def create_app():
    app = Flask(__name__)
    app.config.from_object(Config)
    db.init_app(app)
    ma.init_app(app)

    @app.route('/books', methods=['GET'])
    def get_books():
        try:
            book_service = BookService()
            books = book_service.get_books()
            return jsonify(books)
        except Exception as e:
            logger.error(f'Error getting books: {e}')
            return jsonify({'error': 'Failed to get books'}), 500

    @app.route('/books', methods=['POST'])
    def create_book():
        try:
            data = request.get_json()
            book_service = BookService()
            book = book_service.create_book(data)
            return jsonify(book), 201
        except Exception as e:
            logger.error(f'Error creating book: {e}')
            return jsonify({'error': 'Failed to create book'}), 500

    @app.route('/books/<int:book_id>', methods=['GET'])
    def get_book(book_id):
        try:
            book_service = BookService()
            book = book_service.get_book(book_id)
            if book:
                return jsonify(book)
            else:
                return jsonify({'error': 'Book not found'}), 404
        except Exception as e:
            logger.error(f'Error getting book: {e}')
            return jsonify({'error': 'Failed to get book'}), 500

    @app.route('/books/<int:book_id>', methods=['PUT'])
    def update_book(book_id):
        try:
            data = request.get_json()
            book_service = BookService()
            book = book_service.update_book(book_id, data)
            if book:
                return jsonify(book)
            else:
                return jsonify({'error': 'Book not found'}), 404
        except Exception as e:
            logger.error(f'Error updating book: {e}')
            return jsonify({'error': 'Failed to update book'}), 500

    @app.route('/books/<int:book_id>', methods=['DELETE'])
    def delete_book(book_id):
        try:
            book_service = BookService()
            book_service.delete_book(book_id)
            return jsonify({'message': 'Book deleted successfully'})
        except Exception as e:
            logger.error(f'Error deleting book: {e}')
            return jsonify({'error': 'Failed to delete book'}), 500

    return app

if __name__ == '__main__':
    app = create_app()
    app.run(debug=Config.DEBUG)
```

**models.py:**
```python
from app import db

class Book(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(100), nullable=False)
    author = db.Column(db.String(100), nullable=False)

    def __init__(self, title, author):
        self.title = title
        self.author = author

    def __repr__(self):
        return f'Book({self.title}, {self.author})'
```

**schemas.py:**
```python
from marshmallow import Schema, fields

class BookSchema(Schema):
    id = fields.Int(dump_only=True)
    title = fields.Str(required=True)
    author = fields.Str(required=True)
```

**repository.py:**
```python
from app import db
from models import Book

class BookRepository:
    def get_books(self):
        return Book.query.all()

    def get_book(self, book_id):
        return Book.query.get(book_id)

    def create_book(self, data):
        book = Book(title=data['title'], author=data['author'])
        db.session.add(book)
        db.session.commit()
        return book

    def update_book(self, book_id, data):
        book = Book.query.get(book_id)
        if book:
            book.title = data['title']
            book.author = data['author']
            db.session.commit()
            return book
        else:
            return None

    def delete_book(self, book_id):
        book = Book.query.get(book_id)
        if book:
            db.session.delete(book)
            db.session.commit()
        else:
            return None
```

**services.py:**
```python
from repository import BookRepository
from schemas import BookSchema

class BookService:
    def __init__(self):
        self.repository = BookRepository()
        self.schema = BookSchema()

    def get_books(self):
        books = self.repository.get_books()
        return [self.schema.dump(book) for book in books]

    def get_book(self, book_id):
        book = self.repository.get_book(book_id)
        if book:
            return self.schema.dump(book)
        else:
            return None

    def create_book(self, data):
        book = self.repository.create_book(data)
        return self.schema.dump(book)

    def update_book(self, book_id, data):
        book = self.repository.update_book(book_id, data)
        if book:
            return self.schema.dump(book)
        else:
            return None

    def delete_book(self, book_id):
        self.repository.delete_book(book_id)
```

**utils.py:**
```python
import logging

class Logger:
    def __init__(self):
        self.logger = logging.getLogger(__name__)
        self.logger.setLevel(logging.INFO)
        handler = logging.FileHandler('logs/backend.log')
        handler.setFormatter(logging.Formatter('%(asctime)s - %(name)s - %(levelname)s - %(message)s'))
        self.logger.addHandler(handler)

    def error(self, message):
        self.logger.error(message)
```
